\section{
  \title{HTML Classes}

  HTML classes used to style the templates.

  \deflist{
    \term{\c{codex-document}}
    \def{The class of the element that contains the whole document in a
      single-file HTML document.}

    \term{\c{codex-section}}
    \def{The class of the element that contains the contents of a section in a
      multi-file HTML document.}

    \term{\c{codex-document-title}}
    \def{In a single-file document, the class of the document's title.}

    \term{\c{codex-section-title}}
    \def{In a multi-file document, the class of the section's title.}

    \term{\c{codex-function}, \c{codex-macro}, \c{codex-generic-function},
      \c{codex-method}}
    \def{The \c{div} that contains the documentation the documentation of one of
      those nodes.}

    \term{\c{codex-variable}}
    \def{The \c{div} that contains the documentation of a variable.}

    \term{\c{codex-struct}, \c{codex-class}, \c{codex-type}}
    \def{The \c{div} that contains the documentation of one of those nodes.}

    \term{\c{codex-docstring}}
    \def{The \c{div} that contains the parsed documentation string of any node.}

    \term{\c{codex-lambda-list}}
    \def{The node that contains a lambda list, which is just text.}
  }
}
